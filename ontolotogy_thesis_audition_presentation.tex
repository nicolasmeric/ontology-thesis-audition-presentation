    % for pdftitle and pdfauthor from \title and \author.
    % An optional argument for \title and \author is supported (class amsart).
    % See hyperref documentation file manual.pdf,
    % Version September 2018, section 5.20, page 30.
    pdfusetitle,
    % The PDF format allows two encodings for bookmarks and entries
    % in the information dictionary: PDFDocEncoding and Unicode as UTF-16BE.
    % Option "pdfencoding" selects between these encodings:
    % • "pdfdoc" uses PDFDocEncoding. It uses just one byte per character,
    %   but the supported characters are limited (244 in PDF-1.7).
    % • "unicode" sets Unicode. It is encoded as UTF-16BE.
    %   Two bytes are used for most characters, surrogates need four bytes.
    % • "auto" PDFDocEncoding if the string does not contain
    %   characters outside the encoding and Unicode otherwise.
    % See hyperref documentation file manual.pdf,
    % Version September 2018, section 5.18, page 30.
    pdfencoding=auto,
    % LaTeX's NFSS is used to assist the conversion of arbitrary TeX strings
    % to PDF strings (bookmarks, PDF information entries).
    % Many math command names (\geq, \notin, ...) are not in control of NFSS,
    % therefore they are defined with prefix 'text' (\textgeq, \textnotin, ...).
    % They can be mapped to short names during the processing to PDF strings.
    % The disadvantage is that they are many hundreds macros
    % that need to be redefined for each PDF string conversion.
    % Therefore this can be enabled or disabled as option 'psdextra'.
    % On default the option is turned off (set to 'false').
    % Turning the option on means that the short names are available.
    % Then \geq can directly be used instead of \textgeq.
    % See hyperref documentation file manual.pdf,
    % Version September 2018, section 5.5, page 23.
    psdextra,
    % Colors the text of links and anchors.
    % The colors chosen depend on the the type of link.
    % At present the only types of link distinguished are citations,
    % page references, URLs, local file references, and other links.
    % Unlike colored boxes, the colored text remains when printing.
    % See hyperref documentation file manual.pdf,
    % Version September 2018, section 3.5, page 10.
    colorlinks=true,
    % Color for normal internal links.
    % See hyperref documentation file manual.pdf,
    % Version September 2018, section 3.5, page 10.
    linkcolor=darkblue,
    % Color for bibliographical citations in text.
    % See hyperref documentation file manual.pdf,
    % Version September 2018, section 3.5, page 10.
    citecolor=darkred,
    % Color for linked URLs.
    % See hyperref documentation file manual.pdf,
    % Version September 2018, section 3.5, page 10.
    urlcolor=darkblue,
  },

  %  %
  %  % Or dvipsnames option can be set by the document class definition.
  %  dvipsnames
  %},
  % As pointed out in Section 5.6.1,
  % measuring the default font size in points is not
  % really a good idea for presentations.
  % Nevertheless, beamer does just that,
  % setting the default font size to 11pt as usual.
  % This may seem ridiculously small,
  % but the actual size of each frame size is by default just 128mm by 96mm
  % and the viewer application enlarges the font.
  % By specifying a default font size smaller than 11pt
  % you can put more onto each slide,
  % by specifying a larger font size you can fit on less.

  % version 3.55, section 18.2.1, page 199.

  % The size of a frame is actually the "paper size" of a beamer presentation,
  % and it is variable.
  % By default, it amounts to 128mm by 96mm.
  % The aspect ratio of this size is 4:3,
  % which is exactly what most beamers offer these days.
  % It is the job of the presentation program

\section{Expérience en matière de recherche}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Proposition de stage}
  Où :

  au sein d'une tâche du projet \glsfirst{SVA}
  qui vise à définir des méthodologies et des outils
  pour la validation des systèmes de véhicules autonomes.

  \medskip

  Objectif :

  Étudier la pertinence des logiques de description
  (ontologies)
  et leur potentielle utilisation dans la définition des concepts
  de sûreté de fonctionnement.

  \medskip

  Idée :

  Orienter la génération de scénarios
  et identifier leurs classes

  \(\implies\) S'assurer de la pertinence des scénarios générés

  \(\implies\) Optimisation de la couverture de test.
  \note{
    La sémantique des arbres de défaillances
    et la modélisation événementielle (AltaRica)
    ne permettent pas d'appréhender la complexité de l'environnement,
    c'est-a-dire la dynamique de l'environnement des systèmes autonomes.
  }
\end{frame}

  \alert{Utilisation des ontologies pour la génération de scénarios}

  Utilisation des ontologies \gls{OWL} pour définir des éléments représentatifs de l'environnement du \gls{VA}.

  Création d'un graphe de dépendance
  entre les éléments de l'ontologie
  afin d'établir des relations additionnelles
  notamment des relations probabilistes et mathématiques.

  \alert{Résultats}

  Gestion difficile du graphe complexe
  associée à une ontologie riche et détaillée.

  Impossibilité de maîtriser la génération
  de scénarios redondants ou superflus.

  \note{%
    Note de Romain :

    le but du stage de Romain était de construire
    un modèle dynamique en AltaRica en intégrant
    des boucles à la modélisation,
    et donc de prendre en compte l'aspect dynamique
    (on donne les consignes de sortie du \gls{TJC}
    en entrée des paramètres du modèle).

  Problème d'explosion combinatoire liée à la complexité de l'environnement.

  Limitations dans le pouvoir d'expressivité.

  Impossibilité d'automatiser la vérification de
  la cohérence des concepts.

  % \defbibfilter{<name>}{<expression>}

  % to be used via the filter option of \printbibliography.
  % The <expression> is a complex test based on the logical operators

  % See biblatex documentation file biblatex.pdf,
  % 3.12, 2018/10/30, section 3.7.9, page 92.
  \defbibfilter{online-and-videos}{%
    type=online or type=video
  }

  % Add printed references bibliography.
